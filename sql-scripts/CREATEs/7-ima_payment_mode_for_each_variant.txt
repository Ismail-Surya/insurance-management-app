CREATE TABLE ima_payment_mode_for_each_variant (
	variant_id INT NOT NULL,
    payment_mode_id INT NOT NULL,
    PRIMARY KEY (variant_id, payment_mode_id),
    FOREIGN KEY (variant_id) REFERENCES ima_product_plan_variant(variant_id),
    FOREIGN KEY (payment_mode_id) REFERENCES ima_payment_mode(payment_mode_id)
);
